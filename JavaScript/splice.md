```js
// 초기 배열 생성
let fruits = ["사과", "바나나", "체리", "딸기", "블루베리"];

// splice() 메소드를 사용하여 배열의 특정 부분을 제거하거나 교체하거나 새 요소를 추가할 수 있습니다.

// fruits 배열에서 인덱스 1(바나나)부터 2개의 요소를 제거합니다.
fruits.splice(1, 2);
console.log(fruits); // 출력: ["사과", "딸기", "블루베리"]

// 인덱스 1(딸기)에 "오렌지"와 "키위"를 추가합니다.
fruits.splice(1, 0, "오렌지", "키위");
console.log(fruits); // 출력: ["사과", "오렌지", "키위", "딸기", "블루베리"]

// 인덱스 2(키위)의 요소를 "망고"로 교체합니다.
fruits.splice(2, 1, "망고");
console.log(fruits); // 출력: ["사과", "오렌지", "망고", "딸기", "블루베리"]
```
